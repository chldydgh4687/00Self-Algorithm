### 실패
```c
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>
int main() {
	int k, n, swap, count = 0;
	int *n_num = NULL;
	while (1) {
		scanf("%d %d", &k, &n);
		if (k == -1 && n == -1) break;

		n_num = (int*)malloc(n * sizeof(int));
		for (int i = 0; i < n; i++) scanf("%d", &n_num[i]);
		for (int i = 0; i < n; i++) {
			for (int j = i + 1; j < n; j++) {
				if (n_num[j] < n_num[i]) {
					swap = n_num[i];
					n_num[i] = n_num[j];
					n_num[j] = swap;
				}
			}
		}
		for (int i = 1; i < n - 1; i++) {
			if ((n_num[i - 1] >= (n_num[i] - k)) && (n_num[i + 1] < (n_num[i] + k))
				|| (n_num[i - 1] > (n_num[i] - k)) && (n_num[i + 1] <= (n_num[i] + k))) {
				count++;
				i++;
			}
			else {
				if (i == 1 || i == n - 2)
					++count;
				++count;
			}
		}
		printf("%d\n", count);
		count = 0;
		free(n_num);
	}
	return 0;
}
```
