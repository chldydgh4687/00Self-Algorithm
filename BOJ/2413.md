 #include <stdio.h>
 #include <stdlib.h>
 int main(){
     int n; scanf("%d", &n);
     int *ar = (int*)malloc(n*sizeof(int));
     int *count = (int*)malloc(n*sizeof(int));
     for(int i = 0; i < n; i++){
         scanf("%d", &ar[i]);
         count[i] = 0;
     }

     int swap;
     for(int i = 0; i < n; i++){
        for(int j = i+1; j < n; j++){
            if(ar[i]-1 == ar[j] && count[j] != 1){
                 swap = ar[i];
                 ar[i] = ar[j];
                 ar[j] = swap;
                 count[j] = 1;
                 break;
            }
        }
     }


    for(int k = 0; k < n; k++){
        printf("%d ",ar[k]);
    }
    free(ar);
    free(count);
 }
